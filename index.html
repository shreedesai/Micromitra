Google Drive / Sheets Backend Setup
Connect a Google Sheet to store all student registrations, quiz results, reflections and roadmaps automatically.
Step 1 — Create Google Apps Script Web App
1
Open Google Apps Script
Go to script.google.com, create a new project and paste the Apps Script code below.
2
Paste this Apps Script code
// NMC CompetencyAI — Google Apps Script Backend // Paste this in Google Apps Script > Code.gs const SHEET_ID = 'YOUR_GOOGLE_SHEET_ID_HERE'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); const sheetName = data.type || 'Registrations'; let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); // Add headers based on type if (sheetName === 'Registrations') { sheet.appendRow(['Timestamp','StudentID','Name','Roll','Batch','Email','Gender','Level']); } else if (sheetName === 'QuizResults') { sheet.appendRow(['Timestamp','StudentID','Name','Roll','Level','Score','Total','Pct','WrongCodes']); } else if (sheetName === 'Reflections') { sheet.appendRow(['Timestamp','StudentID','Name','Roll','ReflectionText','SDLReadiness','PrimaryDomain','Level','Quality','Gaps','Strengths']); } else if (sheetName === 'Roadmaps') { sheet.appendRow(['Timestamp','StudentID','Name','Roll','Level','RoadmapJSON']); } } sheet.appendRow(data.row); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheetName = e.parameter.sheet || 'Registrations'; const sheet = ss.getSheetByName(sheetName); if (!sheet) return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON); const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return ContentService.createTextOutput(JSON.stringify({rows})) .setMimeType(ContentService.MimeType.JSON); }
3
Deploy as Web App
In Apps Script: Deploy → New deployment → Web App → Execute as: Me → Who has access: Anyone → Deploy. Copy the Web App URL.
4
Create Google Sheet
Create a new Google Sheet, copy its ID from the URL (the long string between /d/ and /edit), and paste it in the script above where it says YOUR_GOOGLE_SHEET_ID_HERE.
